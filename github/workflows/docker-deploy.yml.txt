name: Docker CI/CD

on:
  push:
    branches:
      - main

jobs:
  build-and-deploy:
    runs-on: ubuntu-latest

    steps:
      - name: Checkout Code
        uses: actions/checkout@v3

      - name: Set up Docker
        uses: docker/setup-buildx-action@v2

      - name: Log in to Docker Hub
        uses: docker/login-action@v2
        with:
          username: ${{ secrets.DOCKER_USERNAME }}
          password: ${{ secrets.DOCKER_PASSWORD }}

      - name: Build & Push Backend Image
        run: |
          docker build -t ${{ secrets.DOCKER_USERNAME }}/movie-backend:latest ./backend
          docker push ${{ secrets.DOCKER_USERNAME }}/movie-backend:latest

      - name: Build & Push Frontend Image
        run: |
          docker build -t ${{ secrets.DOCKER_USERNAME }}/movie-frontend:latest ./frontend
          docker push ${{ secrets.DOCKER_USERNAME }}/movie-frontend:latest

      - name: SSH & Deploy Containers
        uses: appleboy/ssh-action@v0.1.7
        with:
          host: ${{ secrets.SERVER_IP }}
          username: ubuntu
          key: ${{ secrets.SERVER_SSH_KEY }}
          script: |
            docker pull ${{ secrets.DOCKER_USERNAME }}/movie-backend:latest
            docker pull ${{ secrets.DOCKER_USERNAME }}/movie-frontend:latest
            docker-compose -f ~/movie-streaming/docker-compose.yml up -d
